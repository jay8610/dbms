create table areas (
    radius int,
    area decimal(10, 2),
    diameter int,
    circumference decimal(10, 2)
);

delimiter //
create procedure p_circle()
begin
    declare v_radius int default 5;
    declare v_area decimal(10, 2);
    declare v_diameter int;
    declare v_circumference decimal(10, 2);
    cloop: LOOP
    if v_radius > 9 then
    leave cloop;
    end if;
    set v_area = 3.14 * v_radius * v_radius; 
    set v_diameter = 2 * v_radius;
    set v_circumference = 2 * 3.14 * v_radius;
    insert into areas (radius, area, diameter, circumference)
    values (v_radius, v_area, v_diameter, v_circumference);
    set v_radius = v_radius + 1;
    end loop cloop;
    end;
//
delimiter ;
call p_circle();
select * from areas;
